function calculate() { const v = JSON.parse(document.getElementById('varnishSelect').value); const d = parseFloat(document.getElementById('diameter').value); const l = parseFloat(document.getElementById('length').value); const type = document.getElementById('coatingType').value; // Geometrie const sideArea = Math.PI * d * l; const bottomArea = (Math.PI * Math.pow(d, 2)) / 4; const area = (type === 'internal') ? (sideArea + bottomArea) : sideArea; // SCV const scv = (v.sc / 100) * (v.wetD / v.dryD) * 100; // Výpočet nánosů (v mg) const getWet = (thick) => (thick / 10000) * (100 / scv) * v.wetD * (area/100) * 1000; const getDry = (thick) => (thick / 10000) * v.dryD * (area/100) * 1000; const t = translations[currentLang] || translations['cs']; document.getElementById('resContent').innerHTML = `
${t.area}
${(area/100).toFixed(2)} cm²
SCV: ${scv.toFixed(2)}%
SPECIFIKACE NÁNOSU (Target Range)
MOKRÝ (Min ${v.minT}μ) ${getWet(v.minT).toFixed(1)}mg
SUCHÝ (Min ${v.minT}μ) ${getDry(v.minT).toFixed(1)}mg
MOKRÝ (Max ${v.maxT}μ) ${getWet(v.maxT).toFixed(1)}mg
SUCHÝ (Max ${v.maxT}μ) ${getDry(v.maxT).toFixed(1)}mg
`; const resultsCard = document.getElementById('resultsCard'); resultsCard.style.display = 'block'; resultsCard.scrollIntoView({ behavior: 'smooth' }); }
